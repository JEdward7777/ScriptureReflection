|ACT 2:1|پېنتېكوست كۈنى تولۇق كەلگەندە، بارلىق ئادەملەر بىر ھالەتتە، بىر ئورنىدا جەم بولغان.||
|ACT 2:2|كۆكتىن كۈتۈشسىز كۈچلۈك شامال قاققاندەك بىر ئاۋاز كەلدى، شۇ ئاۋاز ئۇلار ئولتۇرغان ئۆزنى تولدۇردى.||
|ACT 2:3|ئوت يالقانلىق تىللار ئۇلارغا كۆرۈنگەن، ھەر بىرىنىڭ ئۈستىگە تارقىلىپ تۇتاشتى.||
|ACT 2:4|ئۇلارنىڭ ھەممىسى مۇقەددەس روھ بىلەن تولۇپ كەتتى، روھ ئۇلارغا سۆز دېگەنچە، باشقا تىللار بىلەن سۆزلەپ كەتتى.||
|ACT 2:5|بۇ ۋاقىتتا، يېرۇسالېمدا، دۇنيا بويىچە ھەر قايسى جايلاردىن كەلگەن كۆپ ئىيمانلىق يەھۇدىي كىشىلەر ھام ياشايۋاتىدۇ.||
|ACT 2:6|بۇ ئاۋازنى ئاڭلىغاندا، خەلق شۇ جايغا يىغىلىپ كەتتى ۋە ھەر بىرى ئېتىقادچىلارنىڭ ئۆز تىلىدا سۆزلەۋاتقانلىقىنى ئاڭلاپ، ھەيران بولدى.||
|ACT 2:7|ئۇلار ھەيران بولۇپ تەۋرە كەتتى: قاراڭ، سۆزلىگەنلەرنىڭ ھەممىسى گالىلىيەلىكلەرمۇ؟||
|ACT 2:8|ئۇلار قانداق ئىشقا كەلدى كى، بىزنىڭ ھەر بىرىمىز ئۆز تىلىمىزدا، يەنى تولغان تىلىمىزدا ئاڭلاۋاتىشىمىز؟||
|ACT 2:9|بىزنىڭ ئارىدا پارتىيالىق، مېدىيالىق، ئېلاملىق ئادەملەر، شۇنداقلا مېسوپوتامىيەدىكىلەر، يەھۇدىيەدىكىلەر، كاپادوكىيادىكىلەر، پونتۇستا، ئاسىيادىكىلەر بار||
|ACT 2:10-11|فرىگىيە ۋە پامفىلىيەدىن، مىسىر، لىۋىيەنىڭ كۈرەن يېقىنىدىن كەلگەنلەر، بۇ يەردە مېيمان بولۇۋاتقان رىملىقلار ــ يەھۇدىى بولسۇن، دىنىي تەرەپتىن كىرىپ كەتكەنلەر بولسۇن، كرېتلار ياكى ئەرەبلەر بولسۇن، بىز ھەممىمىز ئۇلارنىڭ خۇدا قىلغان چوڭ ئىشلىرىنى بىزنىڭ ئۆز تىلىمىزدا سۆزلەۋاتقانلىقىنى ئاڭلاۋاتىمىز!||
|ACT 2:12|ئۇلار ھەممىسى ھەيران قالدى ۋە شاھشات بولدى، بىر-بىرىگە: 'بۇ نېمە بولۇشى مۇمكىن؟' دەپ سۆزلەشتى.||
|ACT 2:13|ئەمما بەزى كىشىلەر: 'بۇلار يېڭى شاراب ئىچىپ مەست بولۇپ كەتتى!' دەپ ھەجۋى قىلدى.||
|ACT 2:14|ئەمما پېتېر باشقا ئون بىر كىشى بىلەن بىرگە تۇرۇپ، ئاۋازىنى بالدۇر قىلىپ، خەلقنىڭ ئالدىدا دېدى: «ئەي يەھۇدىيەدىكىلەر ۋە يېرۇسالېمدا ياشاۋاتقانلار! بۇ نەرسە سىلەرگە ئاشكارا بولسۇن، سۆزلىرىمنى قىزغىن ئەستە تۇتۇڭلار!»||
|ACT 2:15|بۇلار سىلەر ئويلىغاندەك مەست ئەمەس، چۈنكى ھازىر تەك ئەتىگەن سائەت توققۇز بولدى.||
|ACT 2:16|ئەمما بۇ، يوئېل پەيغەمبەر ئارقىلىق ئالدىن دەپ كەتكەن نەرسىدۇر:||
|ACT 2:17|خۇدا دېدى: «مەن ئاخىرقى كۈنلەردە روھىمنى ھەممە خەلق ئۈستىگە سۇيامەن؛ سىلەرنىڭ ئوغۇل-قىزلىرىڭلار پەيغەمبەرلىك قىلىپ سۆزلەيدۇ، سىلەرنىڭ يىگىتلىرىڭلار كۆزگە كەلگەن ئالامەتلەرنى كۆرۈشچە بولىدۇ؛ سىلەرنىڭ قېرىلىرىڭلار ئالامەت چۈش كۆرۈشچە بولىدۇ؛||
|ACT 2:18|شۇ ۋاقىتلarda، مەننىڭ ئادەملىرىمغا، ئەرلەرگەم ئاياللارغا، رۇھىمىنى سۇيىمەن، ئۇلار پەيغەمبەرلىك قىلىدۇ.||
|ACT 2:19|مەن ئاسمان ئۈستىدە ئەجەب ئىشلارنى، زېمىن تۆۋەندە ئالامەتلەرنى، قان، ئوت، تۈتەك بۇلۇتلارنى كۆرسەتەمەن.||
|ACT 2:20|رەبنىڭ چوڭ ۋە ئەزىز كۈنى كەلمەشتىن بۇرۇن، قۇياش قاراڭغۇ بولىدۇ، ئاي قانغا ئۆزگەرەدۇ.||
|ACT 2:21|شۇ ۋاقىتتا بۇلار ئەمەلگە ئاشىدۇ: رەبنىڭ ئىسمى بىلەن چاقىرىنغان ھەر قانداق كىشى قۇتقۇزۇلىدۇ.||
|ACT 2:22|ئىسرائىللار، بۇ سۆزلەرنى ئاڭلاڭلار. ناسارەتلىق ئەيسا، خۇدا سىزلەرگە كۈچلۈك ئىشلەر، كارامەتلەر ۋە مۇشەننەتلەر بىلەن ئىسپاتلىغان بىر كىشى، بۇ ئىشلار سىزلەرگە تولا مەلۇم.||
|ACT 2:23|ئۇ كىشىنى خۇدا ئالدىن بەلگىلىگەن پىلان ۋە بىلىم بويىچە سىلەرگە تەسلىم قىلغان، سىلەر قانۇنسىز كىشىلەر ئارقىلىق ئۇنى سەللىب ئۆلتۈرگەنلەر.||
|ACT 2:24|ئەمما خۇدا ئۇنى ئۆلۈم ئازابىدىن قۇتۇلدۇرۇپ، يەنە تىرىلدۈردى. چۈنكى ئۆلۈم ئۇنى ھېچقانداق تۇتۇپ تۇرالمىدى.||
|ACT 2:25|داۋۇت زەبۇردا بۇ ھەقتە مۇنداق دەپ ئالدىن سۆزلەپ كەتكەن: «مەن پەرۋەردىگارسا ھەمىشە كۆزۈمنىڭ ئالدىدا تۇتۇپ كېلىۋاتىمەن؛ ئۇ مەنمۇنچە ئوڭ تەرەپتە تۇرغاچقا، مەن ھېچقانداق تەۋرەنمەيمەن.»||
|ACT 2:26|شۇ ئۈچۈن يۈرەكىم شاد بولدى، گەپىم خۇشاللىق بىلەن سۆزلەيدۇ؛ بەدىنىم ئۈمىد ئىچىدە تۇرۇپ قالىدۇ.||
|ACT 2:27|چۈنكى سەن جېنىمنى جەھەنمەتتە قالدۇرمايسەن، سېنىڭ مۇقەددەس كىشىڭگە چىرىشمىنى كۆرسەتمەيسەن.||
|ACT 2:28|سەن ماڭا ھايات يوللىرىنى كۆرسەتتىڭ؛ سىز بىلەن بىلەشكەندە مېنى شادلىق ۋە خۇرامغا تولدۇرساڭ.||
|ACT 2:29|ئى قېرىنداشلار، مەن سىلەرگە داۋۇت پاترىيارخ ھەققىدە ئوچۇق ۋە قەتئىي سۆزلەش ئۈچۈن ھوقۇقىم بار: ئۇ ئۆلگەن ۋە دەف قىلىنغان، ۋە ئۇنىڭ قەبرى بۈگۈنكى كۈنگە قەدەر بىزنىڭ ئىچىدە تۇرۇپتۇ.||
|ACT 2:30|ئۇ پەيغەمبەر بولۇپ، خۇدا ئۇنىڭ پۇشتىدىن كەلگەن كەرپى بىلەن، تەن تەرەپلىك، مەسىھنى قايتا تىرىپ ئۇنىڭ تەختىغا ئولتۇرغۇزۇشقا قەسەم بىلەن ۋەدە بەرگەنلىكىنى تولا بىلىپتۇ.||
|ACT 2:31|ئۇ مەسىھنىڭ ئۆلگەندىن كېيىن قايتا تېرەلگەنلىكىنى ئالدىن بىلگەن ۋە بۇ ئۈچۈن مەسىھنىڭ ئۆلۈك دۇنياسىدا قالدۇرۇلمىغىنى ۋە تېنى چىريمەيدىغىنىنى ئېيتقان.||
|ACT 2:32|خۇدا بۇ ئەيسا نى تىرىپ كەتتى، بىز ھەممىلىرىمىز بۇنىڭ گۇۋاھلىرى بولۇپ تۇرۇپ.||
|ACT 2:33|ئۇ خۇدا تەرەپلىك ئوڭ تەرەپكە يۇقىرى كۆتۈرۈلۈپ، ئاتا تەرەپلىك ۋەدە قىلغان مۇقەددەس روھنى ئالغان ۋە شۇنىڭ بىلەن ھازىر سىلەر كۆرۈپ تۇرغان ۋە ئاڭلاپ تۇرغان نەرسىلەرنى سۇيۇپ چۈشۈرگەن.||
|ACT 2:34-35|چۈنكى داۋۇت ئەرشكە چىقمىغان، بەلكى ئۇ زەبۇردا دېگەن: «خۇدا مېنىڭ رەببىمگە دېدىكى: «سېنىڭ دۈشمەنلىرىڭنى پاڭلارىڭ ئۈچۈن ئەۋىزلىك قىلىپ قويغۇچە، مېنىڭ ئوڭ تەرەپimdا ئولتۇر!»»||
|ACT 2:36|شۇڭا، ھەممە ئىسرائىل ئادەملىرى، بۇنى ئېنىق بىلەڭلار: خۇدا سىلەر سەللىگەن بۇ ئەيسانى ھەم رەب ھەم مەسىھ قىلغان!||
|ACT 2:37|بۇ سۆزلەرنى ئاڭلىغاندا، ئۇلارنىڭ يۈرەكلىرى چاق نەپەس قالدى ۋە پېترۇس بىلەن باشقا روسۇللارغا: 'ئى قېرىنداشلار، بىز نېمە قىلىش كېرەك؟' دەپ سوراشتى.||
|ACT 2:38|پېتېر ئۇلارغا: توبا قىلىڭلار، ھەر بىرىڭلار ئەيسا مەسىھنىڭ ئاتىدا گۇناھلىرىڭلار كەچۈرۈلۈشى ئۈچۈن سۇغلاق قىلىڭلار ۋە بۇنى قىلساڭلار، خۇدا سەۋغىسى بولغان مۇقەددەس روھ سىلەرگە كېلىپ قالىدۇ.||
|ACT 2:39|چۈنكى بۇ ۋەدە سىزلەرگە ۋە سىزلەرنىڭ بالىلىرىڭلارغا، ئۇزاقتا تۇرۇۋاتقانلارغا، يەنى خۇدايىمىز پەرۋەردىگار چاقىرغان ھەممىسىگە بېرىلىدۇ.||
|ACT 2:40|پېترۇس يەنە بەزى باشقا سۆزلەر بىلەن ئۇلارنى ئاڭلاتتى ۋە ئۇلارغا: «ئۆزۈڭلارنى بۇ چىلاپ نەسلمىدىن قۇتۇلدۇرۇڭلار!» دەپ توختىدى.||
|ACT 2:41|ئۇنىڭ سۆزىنى قوبۇل قىلغانلار سۇغا چۆمۈلدۈرۈلدى. شۇ كۈن جامائەتكە تەخمىنەن ئۈچ مىڭ كىشى قوشۇلدى.||
|ACT 2:42|ئۇلار روسۇللارنىڭ ئۆگىتىگە، ئىشەنچلىكلەرنىڭ بىرلىك-دوستلۇقىغا، نان سۇرۇشقا ۋە دۇئا قىلىشقا تولا مەشغۇل بولدى.||
|ACT 2:43|ھەم ھەر بىر كىشىنىڭ يۈرەكلىرىدە ھەيرانلىق ۋە ئەجەبلەش پەيدا بولدى، ۋە روسۇللار تەرىپىدىن كۆپ مۇجىزە ۋە ئالامەت ئىشلەپ چىقىرىلدى.||
|ACT 2:44|ھەممە ئىشىنىپ قالغانلار داۋام بىرگە تۇرۇپ، بارلىق نەرسىنى ئورتاق ئىشلەپ كەتتى.||
|ACT 2:45|ئۇلار مال-مۈلۈكلىرىنى ساتتى، پۇلنى ھەر كىشىنىڭ ئېھتىياجى بويىچە ھەممىگە تەقسىم قىلدى||
|ACT 2:46|ئۇلار ھەر كۈنى ئىبادەتخانادا بىرگە توپلانىپ تۇرۇشاتتى، ئۆيلەردە شادلىق ۋە چىن يۈرەك بىلەن ئورتاق تاماق يېيىپ، ناننى سۇزۇپ يېگەندە،||
|ACT 2:47|ئۇلار خۇداگە پەزەل دەپ تۇراتتى ۋە ھەر كىشى بىلەن جاھلىق بولدى. پەرۋەردىگار ھەر كۈنى قۇتۇلۇۋاتقانلارنى توپلۇققا قوشۇپ كەتتى.||
